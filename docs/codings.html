  function flat(arr,flatted){
  if(!flatted){var flatted =[];}
  for(var i = 0; i< arr.length; i++){   
    if(Array.isArray(arr[i]) ){     
       flat(arr[i],flatted);     
    }else{      
      flatted.push(arr[i].toString());    
    }
}
return flatted;
}
